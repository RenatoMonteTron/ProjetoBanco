--Criando Tabelas

-- Criando Tabela CategoriaProduto
CREATE TABLE TBLCATEGORIAS (
    Id INT PRIMARY KEY,
    Nome VARCHAR(100) NOT NULL UNIQUE
);

-- Criando Tabela Perfil
CREATE TABLE TBLPERFIS (
    Id INT PRIMARY KEY,
    Nome VARCHAR(100) NOT NULL UNIQUE
);

-- Criando Tabela Pagamento
CREATE TABLE TBLPAGAMENTOS (
    Id INT PRIMARY KEY,
    Descricao VARCHAR(100) NOT NULL UNIQUE
);

-- Criando Tabela Permissoes
CREATE TABLE TBLPERMISSOES (
    Id INT PRIMARY KEY,
    Permissao VARCHAR(100) NOT NULL
);

-- Criando Tabela Permissaoxperfil
CREATE TABLE TBLPERMISSAOXPERFIL (
    Id INT PRIMARY KEY,
    PermissaoId INT NOT NULL,
    PerfilId INT NOT NULL,
    CONSTRAINT fk_usuario_permissao FOREIGN KEY (PermissaoId) 
    REFERENCES TBLPERMISSOES(Id) ON DELETE CASCADE,
    CONSTRAINT fk_usuario_perfil FOREIGN KEY (PerfilId) 
    REFERENCES TBLPERFIS(Id) ON DELETE CASCADE
);

-- Criando Tabela Fornecedor
CREATE TABLE TBLFORNECEDORES (
    Id INT PRIMARY KEY,
    Nome VARCHAR(100) NOT NULL,
    CNPJ VARCHAR(20) NOT NULL UNIQUE,
    Telefone VARCHAR(20),
    Email VARCHAR(100),
    Endereco BLOB SUB_TYPE TEXT
);

-- Criando Tabela Usuario
CREATE TABLE TBLUSUARIOS (
    Id INT PRIMARY KEY,
    Nome VARCHAR(100) NOT NULL,
    Cpf VARCHAR(14) NOT NULL UNIQUE,
    Cargo VARCHAR(50),
    PerfilId INT NOT NULL,
    Senha VARCHAR(255) NOT NULL,
    CONSTRAINT fk_perfil_usuario FOREIGN KEY (PerfilId)
    REFERENCES TBLPERFIS(Id)
);

-- Criando Tabela Cliente
CREATE TABLE TBLCLIENTES (
    Id INT PRIMARY KEY,
    Nome VARCHAR(100) NOT NULL,
    CPF VARCHAR(14) NOT NULL UNIQUE,
    Endereco BLOB SUB_TYPE TEXT,
    Telefone VARCHAR(20),
    Email VARCHAR(100),
    Dtnasc DATE,
    DataCadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Criando Tabela Produto
CREATE TABLE TBLPRODUTOS (
    Id INT PRIMARY KEY,
    Codigo VARCHAR(50) NOT NULL UNIQUE,
    Nome VARCHAR(100) NOT NULL,
    Descricao BLOB SUB_TYPE TEXT,
    Custo DECIMAL(10,2) NOT NULL,
    ValorVenda DECIMAL(10,2) NOT NULL,
    CategoriaId INT NOT NULL,
    FornecedorId INT NOT NULL,
    CONSTRAINT fk_produto_categoria FOREIGN KEY (CategoriaId) REFERENCES TBLCATEGORIAS(Id),
    CONSTRAINT fk_produto_fornecedor FOREIGN KEY (FornecedorId) REFERENCES TBLFORNECEDORES(Id)
);

-- Criando Tabela Estoque
CREATE TABLE TBLESTOQUE (
    Id INT PRIMARY KEY,
    ProdutoId INT NOT NULL UNIQUE,
    Localizacao VARCHAR(50) NOT NULL,
    Quantidade INT NOT NULL
);

-- Criando Tabela VendasTot
CREATE TABLE TBLVENDASTOT (
    Id INT PRIMARY KEY,
    Dt DATE NOT NULL,
    Hr TIME NOT NULL,
    ValorTotal DECIMAL(10,2) NOT NULL,
    Desconto DECIMAL(10,2) DEFAULT 0.00,
    Comissão DECIMAL(10,2) DEFAULT 0.00,
    ClienteId INT NOT NULL,
    PagamentoId INT NOT NULL,
    VendedorId INT NOT NULL,
    CONSTRAINT fk_vendasTot_cliente FOREIGN KEY (ClienteId) 
    REFERENCES TBLCLIENTES(Id),
    CONSTRAINT fk_vendasTot_vendedor FOREIGN KEY (VendedorId) 
    REFERENCES TBLUSUARIOS(Id),
    CONSTRAINT fk_vendasTot_pagamento FOREIGN KEY (PagamentoId) 
    REFERENCES TBLPAGAMENTOS(Id)
);

-- Criando Tabela VendasUnicas
CREATE TABLE TBLVENDASUNICAS (
    Id INT PRIMARY KEY,
    VendasTotId INT NOT NULL,
    ProdutoId INT NOT NULL,
    Quantidade INT NOT NULL,
    Desconto DECIMAL(10,2) DEFAULT 0.00,
    Valor DECIMAL(10,2) NOT NULL,
    CONSTRAINT fk_venda_produto FOREIGN KEY (ProdutoId) 
    REFERENCES TBLPRODUTOS(Id),
    CONSTRAINT fk_venda_vendatot FOREIGN KEY (IdVendasTot) 
    REFERENCES TBLVENDASTOT(Id) ON DELETE CASCADE
);

-- Procedure para pesquisar todas as vendasUnicas ligadas à uma vendaTot
CREATE PROCEDURE PesquisarVenda(p_idVendaTot INT)
AS
BEGIN
    FOR SELECT v.Id, v.IdVendasTot, p.Nome AS TBLPRODUTOS, v.Quantidade, v.Valor, v.Desconto
    FROM TBLVENDASUNICAS v
    JOIN TBLPRODUTOS p ON v.ProdutoId = p.Id
    WHERE v.IdVendasTot = :p_idVendaTot
    INTO :Id, :IdVendasTot, :Produto, :Quantidade, :Valor, :Desconto
    DO SUSPEND; 
END

-- Trigger para criar um estoque de um produto assim que cria o produto
CREATE TRIGGER InserirNoEstoque
AFTER INSERT ON TBLPRODUTOS
AS
BEGIN
    INSERT INTO Estoque (Id, ProdutoId, Localizacao, Quantidade)
    VALUES (NEW.ID, NEW.Id, 'Estoque Geral', 1);
END

END

-- Trigger para redusir estoque apos uma venda
CREATE TRIGGER AtualizarEstoque_AposVenda
AFTER INSERT ON TBLVENDASUNICAS
AS
DECLARE V_ESTOQUE INT;
BEGIN
    -- Verifica o estoque atual
    SELECT Quantidade FROM Estoque 
    WHERE ProdutoId = NEW.ProdutoId
    INTO :V_ESTOQUE;

    -- Se não houver estoque suficiente, lança um erro
    IF (V_ESTOQUE < NEW.Quantidade) THEN
    BEGIN
        EXCEPTION ESTOQUE_INSUFICIENTE;
    END

    -- Atualiza o estoque
    UPDATE Estoque 
    SET Quantidade = Quantidade - NEW.Quantidade
    WHERE ProdutoId = NEW.ProdutoId;
END
